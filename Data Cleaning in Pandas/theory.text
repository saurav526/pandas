1Ô∏è‚É£ isnull() ‚Äì Detect missing values
print(df.isnull())

2Ô∏è‚É£ notnull() ‚Äì Detect non-missing values
print(df.notnull())

3Ô∏è‚É£ sum() with isnull() ‚Äì Count missing values
print(df.isnull().sum())

4Ô∏è‚É£ dropna() ‚Äì Remove rows with missing values
df_dropna = df.dropna()
print(df_dropna)

5Ô∏è‚É£ dropna(axis=1) ‚Äì Remove columns with missing values
df_dropna_col = df.dropna(axis=1)
print(df_dropna_col)

6Ô∏è‚É£ fillna() ‚Äì Fill missing values
üîπ Fill with fixed value
df_fill = df.fillna(0)
print(df_fill)

üîπ Fill with mean (numeric columns)
df["Age"].fillna(df["Age"].mean(), inplace=True)
df["Marks"].fillna(df["Marks"].mean(), inplace=True)
print(df)

üîπ Fill with mode (categorical)
df["City"].fillna(df["City"].mode()[0], inplace=True)
print(df)

7Ô∏è‚É£ replace() ‚Äì Replace incorrect values
df_replace = df.replace("Delhi", "New Delhi")
print(df_replace)

8Ô∏è‚É£ duplicated() ‚Äì Detect duplicate rows
print(df.duplicated())

9Ô∏è‚É£ drop_duplicates() ‚Äì Remove duplicate rows
df_nodup = df.drop_duplicates()
print(df_nodup)

üîü astype() ‚Äì Change data type
df["Age"] = df["Age"].astype(int)
print(df.dtypes)

1Ô∏è‚É£1Ô∏è‚É£ rename() ‚Äì Rename columns

print(df_rename)

1Ô∏è‚É£2Ô∏è‚É£ apply() ‚Äì Apply custom function
df["Marks_Status"] = df["Marks"].apply(lambda x: "Pass" if x >= 40 else "Fail")

1Ô∏è‚É£3Ô∏è‚É£ clip() ‚Äì Limit values
df["Marks"] = df["Marks"].clip(lower=40, upper=100)

1Ô∏è‚É£4Ô∏è‚É£ round() ‚Äì Round values
df["Marks"] = df["Marks"].round(1)
